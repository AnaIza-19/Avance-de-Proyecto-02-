    // Definir Variable a usar

    IMPRIMIR "¿Cuántas notas desea ingresar? (N > 0)"

    MIENTRAS N <= 0 HACER

        LEER N
    FIN_MIENTRAS
    // Registrar notas, y almacenarlas en lista de Notas
    PARA i ← 1 HASTA N HACER
        IMPRIMIR "Ingrese la nota ", i, ": "
        nota ← RegistrarNota()
        lista_notas.Agregar(nota)
    FIN_PARA
    // Calcular el promedio de las notas
    suma ← 0

     PARA i ← 1 HASTA CantidadN (lista_notas) HACER

// Función para ingresar una nota válida (mayor a 0 y hasta 100)
FUNCION RegistrarNota() :
// PRE las notas deben ser mayores a 0, y no deben ser mayores a 100
// POST notas
    LEER nota
    MIENTRAS nota <= 0 o nota > 100 HACER
        IMPRIMIR "La nota no cumple con los requisitos (mayor a 0 y hasta 100). Intente de nuevo: "
        LEER nota
    FIN_MIENTRAS
